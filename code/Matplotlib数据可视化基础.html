x = np.arange(0, 1.1, 0.1)
print(x)

plt.plot(x, x**2)   # 第二环节，绘制图形

plt.ylim(0, 1)

plt.ylabel('y')

plt.rcParams['axes.unicode_minus'] = False

# 标记柱子高度
my_height = values[-1, 3:6]
for i in range(len(my_height)):
    plt.text(i, my_height[i]+1000, my_height[i], va='bottom', ha='center')

plt.show()
